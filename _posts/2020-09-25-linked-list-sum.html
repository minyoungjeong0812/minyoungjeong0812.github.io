There are two ways to do this. First is utilizing separate insert and append function (not leetcode-bound solution). 

class node:
    def __init__(self,data,next=None):
        self.data=data
        self.next=next

class linked_list:
    def __init__(self):
        self.head=None

    def insert(self,data):
        new_node=node(data)
        new_node.next=self.head
        self.head=new_node

    def appending(self,data):
        cur = self.head
        while cur.next:
            cur = cur.next
        cur.next=node(data)

    def printing(self):
        cur=self.head
        while cur.next:
            print(cur.data)
            cur = cur.next
        print(cur.data)

Another way is to follow Leetcode restrictions where we don't neccessarily have append and insert functions for LinkedList.



 


class Solution:

        list1=[]
        list2=[]

        while l1.next:
            list1.append(l1.val)
            l1=l1.next
        list1.append(l1.val)

        while l2.next:
            list2.append(l2.val)
            l2 = l2.next
        list2.append(l2.val)

        int2 =int(','.join([str(x) for x in list2[::-1]]).replace(',',''))

        num_list=[int(d) for d in str(int1+int2)][::-1]

        cur=l3

                cur=cur.next
            cur.next=ListNode(num_list[x])
